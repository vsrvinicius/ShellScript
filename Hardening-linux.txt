#!/bin/bash

echo "Atualizando o sistema..."

sleep 2

apt update

# Install rsyslog

echo "Instalando RSYSLOG"

sleep 3

apt install rsyslog

# Configuring the Log Host Server

# Make a backup before

echo "Criando um backup de /etc/rsyslog.conf"

sleep 3

cp /etc/rsyslog.conf /etc/rsyslog_original.config

echo "module(load="imtcp")
      input(type="imtcp" port="514")" >> /etc/rsyslog.conf

# Add the configuration of location where rsyslog will store your logs

echo "$template remote-incoming-logs, "/var/log/remote/%HOSTNAME%".log
      *.* ?remote-incoming-logs" >> /etc/rsyslog.conf

systemctl restart rsyslog

echo "Allow port 514/tcp on firewall"

sleep 3

ufw allow 514/tcp

echo "Restart the firewall service"

sleep 3

systemctl restart ufw

echo "Configure the logging client"

sleep 3

echo "*.*@@ip:514" >> /etc/rsyslog.conf

echo "Restart RSYSLOG"

sleep 3

systemctl restart rsyslog


echo "You can view the log messages on /var/log/remote ..."

sleep 5

echo "Backup /etc/fstab"

sleep 2

cp -a /etc/fstab /etc/fstab.bak

# Make 3GB file and format it with ext3

echo "Making a 3GB file and format it with ext3"

sleep 2

dd if=/dev/zero of=/var/tempFS bs=1024 count=3072000 /sbin/mkfs.ext3 /var/tempFS

# Create a backup copy of current /tmp

echo "Creating a backup of current /tmp, changing permissions of new directory and copying the data"

sleep 2

rsync -av /tmp/ /tmpbackup/

# Mount our new tmp partition and change permissions

mount -o loop,noexec,nosuid,rw /var/tempFS /tmp
chmod 1777 /tmp

# Copy the old data

rsync -av /tmpbackup/ /tmp/

# Update fstab
# vim /etc/fstab
echo "/tmp ext3 loop,nosuid,noexec,rw 0 0" >> /etc/fstab

sleep 2

# Test your fstab entry

mout -o remount /tmp

# Verify that your /tmp mount is working

df -h

# Should look like this /var/tempFS 962M 18M 896M 2% /tmp

echo "Securing /var/tmp"

sleep 2

# Rename /var/tmp and create a symbolic link to /tmp

mv /var/tmp /var/vartmp
ln -s /tmp /var/tmp

# Copy the old data back

rsync -av /var/vartmp/ /tmp/

# Remove /var/vartmp

rm -rf /var/vartmp

echo "Securing /dev/shm"

sleep 2

# vim /etc/fstab

# Locate: none /dev/shm tmpfs defaults,rw 0 0

echo "/dev/shm tmpfs defaults,nosuid,noexec,rw 0 0" >> /etc/fstab

# Remount /dev/shm

mount -o remount /dev/shm

# Generate a Hashed Password

echo "Securing Grub"

sleep 2

grub-mkpasswd-pbkdf2 | tee passwd

touch teste

grep -Po '^.{32}\K.*' passwd >> teste

rm passwd

passwd="$(cat teste)"

read -p "Escolha o usuario para fazer login no Grub: " user

echo "cat << EOF
set superusers="$user"
password_pbkdf2 $user $passwd
EOF" >> /etc/grub.d/00_header

update-grub

rm teste

echo "Escolha uma senha para o usÃ¡rio Root"

sleep 2

sudo passwd root

echo "Instalando NTP server"

sleep 2


apt install ntp

grep -v '^pool' /etc/ntpsec/ntp.conf > /etc/ntpsec/ntp.conf1

echo "server 0.south-america.pool.ntp.org
server 1.south-america.pool.ntp.org
server 2.south-america.pool.ntp.org
server 3.south-america.pool.ntp.org" >> /etc/ntpsec/ntp.conf1

rm /etc/ntpsec/ntp.conf

mv /etc/ntpsec/ntp.conf1 /etc/ntpsec/ntp.conf

service ntp restart

ufw allow ntp

 ntpdate <server-IP>

service start ntp
